  <main itemscope itemtype="https://schema.org/Recipe">
    <h1 itemprop="name">Frozen Peanut Butter Bites</h1>

    <p><strong>Servings:</strong> <span itemprop="recipeYield">12 bites</span></p>

    <h2>Ingredients</h2>

    <h3>Peanut Butter Banana Base</h3>
    <ul>
      <li itemprop="recipeIngredient">1 mashed ripe banana</li>
      <li itemprop="recipeIngredient">1 cup smooth peanut butter</li>
      <li itemprop="recipeIngredient">1/2 cup plain Greek yogurt</li>
      <li itemprop="recipeIngredient">2 tbsp honey</li>
      <li itemprop="recipeIngredient">1/3 cup chopped cashews</li>
    </ul>

    <h3>Chocolate Topping</h3>
    <ul>
      <li itemprop="recipeIngredient">8 oz dairy-free chocolate chips</li>
      <li itemprop="recipeIngredient">1 tsp coconut oil</li>
      <li itemprop="recipeIngredient">Flake salt, for finishing</li>
    </ul>

    <h2>Instructions</h2>
    <ol>
      <li itemprop="recipeInstructions" itemscope itemtype="https://schema.org/HowToStep">
        <span itemprop="text">Add the mashed banana, peanut butter, Greek yogurt, honey, and chopped cashews to a bowl. Mix until smooth and sticky.</span>
      </li>

      <li itemprop="recipeInstructions" itemscope itemtype="https://schema.org/HowToStep">
        <span itemprop="text">Scoop 12 dollops onto a parchment-lined tray and lightly flatten into disc shapes.</span>
      </li>

      <li itemprop="recipeInstructions" itemscope itemtype="https://schema.org/HowToStep">
        <span itemprop="text">Freeze for 1 hour.</span>
      </li>

      <li itemprop="recipeInstructions" itemscope itemtype="https://schema.org/HowToStep">
        <span itemprop="text">Melt the chocolate chips and coconut oil in the microwave until smooth.</span>
      </li>

      <li itemprop="recipeInstructions" itemscope itemtype="https://schema.org/HowToStep">
        <span itemprop="text">Remove the frozen bites and spoon melted chocolate on top. Sprinkle with flake salt.</span>
      </li>

      <li itemprop="recipeInstructions" itemscope itemtype="https://schema.org/HowToStep">
        <span itemprop="text">Return to the freezer for 15–30 minutes until set. Store in freezer.</span>
      </li>
    </ol>
